\newcommand{\MSE}[1][]  {
    \ifthenelse { \equal{#1}{} }  
        { \ensuremath\mathrm{MSE} }                     % if #1 is blank, use "MSE"
        { \ensuremath\mathrm{MSE}\left(#1\right) }      % if #1 not blank, use "MSE(#1)"
}
\newcommand{\SSE}[1][]  {
    \ifthenelse { \equal{#1}{} }  
        { \ensuremath\mathrm{SSE} }                     % if #1 is blank, use "SSE"
        { \ensuremath\mathrm{SSE}\left(#1\right) }      % if #1 not blank, use "SSE(#1)"
}
